% music-chords.tex — chord symbols (\MusicChord) and Roman numerals.
% Public interface stays template-oriented:
%   \MusicChord{root=E, acc=flat, quality=dom, seven=7, alters={9/flat,5/flat}}
%   \MusicChord{root=C, quality=maj, seven=7, bass=E, bass acc=flat}

\let\savedflat\flat
\let\savedsharp\sharp
\let\savednatural\natural
\usepackage{lilyglyphs}

\providecommand{\RomNum}[1]{\ensuremath{\mathrm{#1}}}
\providecommand{\RomNumS}[2]{\ensuremath{\mathrm{#1}^{#2}}}
\providecommand{\RomViiO}{\ensuremath{\mathrm{vii}^\circ}}

\ExplSyntaxOn

\msg_new:nnnn { musicchord } { missing-root }
  { musicchord:~missing~root. }
  {
    Set~root=<single~letter~A--G>.^^J
    Example~keyval:~root=C,~quality=dom,~seven=7.^^J
    Optional:~acc=flat|sharp.
  }
\msg_new:nnnn { musicchord } { missing-quality }
  { musicchord:~missing~quality. }
  {
    Set~quality=maj|min|dom|aug~(all~lowercase).^^J
    Extensions:~seven=7,~six=6,~etc.
  }
\msg_new:nnnn { musicchord } { unknown-alt-kind }
  { musicchord:~unknown~alter~kind~'#1'. }
  {
    Each~alters~item~must~be~degree/kind.^^J
    kind~must~be~flat,~sharp,~or~natural~(lowercase).^^J
    Example:~alters=\{9/flat,~11/sharp\}.
  }
\msg_new:nnnn { musicchord } { malformed-alt }
  { musicchord:~malformed~alters~item~'#1'. }
  {
    Expected~exactly~one~'/'~separating~degree~and~kind,~e.g.~9/flat~or~13/sharp.
  }
\msg_new:nnnn { musicchord } { unknown-bass-acc }
  { musicchord:~unknown~bass~accidental~'#1'. }
  { Allowed:~bass~acc=flat|sharp|natural,~or~omit~bass~acc. }

\tl_new:N \l__musicchord_root_tl
\tl_new:N \l__musicchord_acc_tl
\tl_new:N \l__musicchord_quality_tl
\tl_new:N \l__musicchord_seven_tl
\tl_new:N \l__musicchord_alters_tl
\tl_new:N \l__musicchord_bass_tl
\tl_new:N \l__musicchord_bassacc_tl
\tl_new:N \l__musicchord_alters_out_tl
\tl_new:N \l__musicchord_input_tl

\clist_new:N \l__musicchord_alters_clist
\seq_new:N \l__musicchord_a_seq
\seq_new:N \l__musicchord_b_seq

\cs_generate_variant:Nn \keys_set:nn { nV }

\cs_new_protected:Npn \__musicchord_keys_define:
  {
    \keys_define:nn { musicchord }
      {
        root .tl_set:N = \l__musicchord_root_tl,
        acc .choices:nn =
          { flat , sharp }
          { \tl_set:Nn \l__musicchord_acc_tl {##1} },
        quality .choices:nn =
          { maj , min , dom , aug }
          { \tl_set:Nn \l__musicchord_quality_tl {##1} },
        seven .tl_set:N = \l__musicchord_seven_tl,
        alters .tl_set:N = \l__musicchord_alters_tl,
        alts .meta:n = { alters = {##1} },
        bass .tl_set:N = \l__musicchord_bass_tl,
        bassacc .tl_set:N = \l__musicchord_bassacc_tl,
        slash .meta:n = { bass = {##1} },
        slashacc .meta:n = { bassacc = {##1} },
      }
  }
\__musicchord_keys_define:

\cs_new_protected:Npn \musicchord_reset:
  {
    \tl_clear:N \l__musicchord_root_tl
    \tl_clear:N \l__musicchord_acc_tl
    \tl_clear:N \l__musicchord_quality_tl
    \tl_clear:N \l__musicchord_seven_tl
    \tl_clear:N \l__musicchord_alters_tl
    \tl_clear:N \l__musicchord_bass_tl
    \tl_clear:N \l__musicchord_bassacc_tl
  }

\cs_new_protected:Npn \musicchord_normalize_input:n #1
  {
    \tl_set:Nn \l__musicchord_input_tl {#1}
    \tl_replace_all:Nnn \l__musicchord_input_tl { bass acc } { bassacc }
    \tl_replace_all:Nnn \l__musicchord_input_tl { slash acc } { slashacc }
  }

\cs_new:Npn \musicchord_letteracc:n #1
  { \mbox { \raisebox { 0.45ex } { \ensuremath {#1} } } }

\cs_new_protected:Npn \musicchord_acc_print:
  {
    \tl_if_empty:NF \l__musicchord_acc_tl
      {
        \exp_args:No \str_case:nn { \tl_use:N \l__musicchord_acc_tl }
          {
            { flat } { \musicchord_letteracc:n { \savedflat } }
            { sharp } { \musicchord_letteracc:n { \savedsharp } }
          }
      }
  }

\cs_new_protected:Npn \musicchord_bassacc_print:
  {
    \tl_if_empty:NTF \l__musicchord_bassacc_tl
      { }
      {
        \exp_args:No \str_case:nnF { \tl_use:N \l__musicchord_bassacc_tl }
          {
            { flat } { \musicchord_letteracc:n { \savedflat } }
            { sharp } { \musicchord_letteracc:n { \savedsharp } }
            { natural } { \musicchord_letteracc:n { \savednatural } }
          }
          {
            \msg_error:nne { musicchord } { unknown-bass-acc }
              { \tl_use:N \l__musicchord_bassacc_tl }
          }
      }
  }

\cs_new_protected:Npn \musicchord_set_alt_tok:nn #1#2
  {
    \str_case:nnF {#2}
      {
        { flat }
          {
            \tl_set:Nx \l_tmpd_tl
              { \exp_not:n { \savedflat \mathrm { #1 } } }
          }
        { sharp }
          {
            \tl_set:Nx \l_tmpd_tl
              { \exp_not:n { \savedsharp \mathrm { #1 } } }
          }
        { natural }
          {
            \tl_set:Nx \l_tmpd_tl
              { \exp_not:n { \savednatural \mathrm { #1 } } }
          }
      }
      {
        \msg_error:nnn { musicchord } { unknown-alt-kind } {#2}
        \tl_clear:N \l_tmpd_tl
      }
  }

\cs_new_protected:Npn \__musicchord_alters_sort:nn #1#2
  {
    \seq_set_split:Nnx \l__musicchord_a_seq { / } { \tl_trim_spaces:n {#1} }
    \seq_set_split:Nnx \l__musicchord_b_seq { / } { \tl_trim_spaces:n {#2} }
    \int_compare:nNnTF
      { \seq_item:Nn \l__musicchord_a_seq { 1 } }
      < { \seq_item:Nn \l__musicchord_b_seq { 1 } }
      { \sort_return_swapped: }
      { \sort_return_same: }
  }

\cs_new_protected:Npn \musicchord_alter_one:n #1
  {
    \tl_set:Nn \l_tmpa_tl {#1}
    \tl_trim_spaces:N \l_tmpa_tl
    \seq_set_split:NnV \l__musicchord_a_seq { / } \l_tmpa_tl
    \int_compare:nNnTF { \seq_count:N \l__musicchord_a_seq } = { 2 }
      {
        \tl_set:Nx \l_tmpb_tl
          { \tl_trim_spaces:n { \seq_item:Nn \l__musicchord_a_seq { 1 } } }
        \tl_set:Nx \l_tmpc_tl
          { \tl_trim_spaces:n { \seq_item:Nn \l__musicchord_a_seq { 2 } } }
        \tl_if_empty:NTF \l_tmpb_tl
          {
            \msg_error:nne { musicchord } { malformed-alt }
              { \tl_use:N \l_tmpa_tl }
          }
          {
            \tl_if_empty:NTF \l_tmpc_tl
              {
                \msg_error:nne { musicchord } { malformed-alt }
                  { \tl_use:N \l_tmpa_tl }
              }
              {
                \exp_args:NVV \musicchord_set_alt_tok:nn \l_tmpb_tl \l_tmpc_tl
                \tl_put_right:NV \l__musicchord_alters_out_tl \l_tmpd_tl
              }
          }
      }
      {
        \msg_error:nne { musicchord } { malformed-alt }
          { \tl_use:N \l_tmpa_tl }
      }
  }

\cs_new_protected:Npn \musicchord_alters_print:n #1
  {
    \clist_clear:N \l__musicchord_alters_clist
    \clist_set:Nx \l__musicchord_alters_clist { \tl_trim_spaces:n {#1} }
    \clist_if_empty:NTF \l__musicchord_alters_clist
      { }
      {
        \clist_sort:Nn \l__musicchord_alters_clist
          { \__musicchord_alters_sort:nn {##1} {##2} }
        \tl_clear:N \l__musicchord_alters_out_tl
        \clist_map_inline:Nn \l__musicchord_alters_clist
          { \musicchord_alter_one:n {##1} }
        \textsuperscript { \ensuremath { ( \tl_use:N \l__musicchord_alters_out_tl ) } }
      }
  }

\cs_new_protected:Npn \musicchord_bass_maybe_print:
  {
    \tl_if_empty:NTF \l__musicchord_bass_tl
      { }
      {
        /
        \tl_use:N \l__musicchord_bass_tl
        \musicchord_bassacc_print:
      }
  }

\cs_new_protected:Npn \musicchord_quality_print:
  {
    \tl_if_empty:NTF \l__musicchord_quality_tl
      { \msg_error:nn { musicchord } { missing-quality } }
      {
        \exp_args:No \str_case:nn { \tl_use:N \l__musicchord_quality_tl }
          {
            { dom }
              {
                \tl_if_empty:NF \l__musicchord_seven_tl
                  { \tl_use:N \l__musicchord_seven_tl }
              }
            { maj }
              {
                \tl_if_empty:NF \l__musicchord_seven_tl
                  { maj \textsuperscript { \tl_use:N \l__musicchord_seven_tl } }
              }
            { min }
              {
                min
                \tl_if_empty:NF \l__musicchord_seven_tl
                  { \textsuperscript { \tl_use:N \l__musicchord_seven_tl } }
              }
            { aug }
              {
                aug
                \tl_if_empty:NF \l__musicchord_seven_tl
                  { \textsuperscript { \tl_use:N \l__musicchord_seven_tl } }
              }
          }
      }
  }

\cs_new_protected:Npn \musicchord_print:
  {
    \tl_if_empty:NTF \l__musicchord_root_tl
      { \msg_error:nn { musicchord } { missing-root } }
      {
        \tl_use:N \l__musicchord_root_tl
        \musicchord_acc_print:
        \musicchord_quality_print:
        \exp_args:NV \musicchord_alters_print:n \l__musicchord_alters_tl
        \musicchord_bass_maybe_print:
      }
  }

\cs_new_protected:Npn \musicchord_chord:n #1
  {
    \musicchord_reset:
    \musicchord_normalize_input:n {#1}
    \keys_set:nV { musicchord } \l__musicchord_input_tl
    \musicchord_print:
  }

\NewDocumentCommand \MusicChord { m }
  {
    \group_begin:
      \musicchord_chord:n {#1}
    \group_end:
  }

\ExplSyntaxOff
