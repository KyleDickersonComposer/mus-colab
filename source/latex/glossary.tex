\newglossaryentry{open-source}{%
  name={open source},
  description={a habit of shipping not only the finished PDF but the actual source files---the ``recipe,'' not just the cake---so anyone can see how the thing was built, suggest a fix, or adapt the material inside the license lines.}
}

\newglossaryentry{version-control}{%
  name={version control},
  description={software that remembers every saved change: what moved, when, and usually who moved it. You stop juggling files named ``final'' and ``final-v3-really'' and start trusting the history instead of your inbox.}
}

\newglossaryentry{github}{%
  name={GitHub},
  description={a hosting site where open projects keep files, conversations, and proposed patches in public. Some people only read; others comment, open discussions, or send suggested edits. Many teams also wire it so a fresh build runs whenever the main branch moves.}
}

\newglossaryentry{plain-text}{%
  name={plain text},
  description={writing stored as ordinary characters---letters, numbers, punctuation---without a proprietary binary wrapper. Humans can read it; diff tools can read it; layout engines can read it. That is why it plays nicely with version history and automated builds.}
}

\newglossaryentry{permissively-licensed}{%
  name={permissively licensed},
  description={released under terms that favor reuse: usually ``keep the credit and pass the license along.'' Still read the actual license file; software lawyers earn their retainers for a reason.}
}
